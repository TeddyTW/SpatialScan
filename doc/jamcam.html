"""Module containing utility functionality to convert Jamcam data into usable format
with the remaining code."""
import pandas as pd
import numpy as np
import requests
import json
import datetime


def get_jamcam_data(
    start_year = 2020,
    start_month = 6,
    start_day = 1,

    end_year = 2020,
    end_month = 6,
    end_day = 2,
):

    start = datetime.datetime(start_year, start_month, start_day)
    end = datetime.datetime(end_year, end_month, end_day)

    days = pd.date_range(start=start, end=end, freq='d')

    ids = pd.read_csv('../data/jamcam_locations.csv')['id']

    res_dict = {}
    num_entries = 0

    for i in range(len(days) - 1):

        start_y, start_m, start_d = days[i].year, days[i].month, days[i].day
        end_y, end_m, end_d = days[i+1].year, days[i+1].month, days[i+1].day

        start_time_string = '{0:04d}-{1:02d}-{2:02d}T00%3A00%3A00.000Z'.format(
            start_y, start_m, start_d
        )
        end_time_string   = '{0:04d}-{1:02d}-{2:02d}T00%3A00%3A00.000Z'.format(
            end_y, end_m, end_d
        )

        for cam_id in ids[:10]:
            data = requests.get("https://urbanair.turing.ac.uk/api/v1/jamcams/raw?camera_id=0000{}".format(cam_id),\
                    "&detection_class=all&starttime={}&endtime={}".format(start_time_string, end_time_string),
                     auth=("admin", "x7WBcuRtrgK8255rPZcB"))

            jreq = json.loads(data.text)
            size = len(jreq)
            if not isinstance(jreq, list):
                continue
            else:
                for j in range(0, size):
                    res_dict[num_entries] = {
                        "camera_id": jreq[j]['camera_id'],
                        "counts": jreq[j]['counts'],
                        "detection_class": jreq[j]['detection_class'],
                        "measurement_start_utc": jreq[j]['measurement_start_utc'],
                    }
                    num_entries += 1

    # Make huge dataframe, then ggregate on hourly
    res_df = pd.DataFrame.from_dict(res_dict, "index")

    return res_df


def format_jamcam_data(jamcam_data: pd.DataFrame, jamcam_locations: pd.DataFrame) -> pd.DataFrame:

    # Convert jamcam data into average hourly counts
    jamcam_data['count'] = jamcam_data['avg'].apply(lambda x: int(360 * x))

    # Fine the dates of which the data spans
    jamcam_data['measurement_end_utc'] = pd.to_datetime(jamcam_data['date_trunc'], dayfirst=True)
    jamcam_data['measurement_start_utc'] = jamcam_data['measurement_end_utc'] - np.timedelta64(1, 'h')

    # Drop useless columns
    ss_df = jamcam_data.drop(["avg", "date_trunc"], axis=1)

    # Merge with location data so that returned dataframe has lon, lat
    ss_df = ss_df.merge(jamcam_locations, how='left', left_on='camera_id', right_on='id')

    # Drop duplicate id column
    ss_df.drop("id", axis=1)

    # re-order columns
    ss_df = ss_df[['camera_id',
                   'lon',
                   'lat',
                   'commonName',
                   'measurement_start_utc',
                   'measurement_end_utc',
                   'detection_class',
                   'count']]
    ss_df.sort_values('measurement_end_utc', inplace=True)

    orig_length = len(ss_df['camera_id'])
    ss_df.drop_duplicates(inplace=True)
    curr_length = len(ss_df['camera_id'])

    print("Dropped {} duplicate rows.".format(orig_length - curr_length))

    # Rename for consitency with further work
    ss_df.rename({'camera_id': 'detector_id'}, axis=1, inplace=True)

    return ss_df
